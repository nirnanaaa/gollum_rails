def validate(record)
  if record.type.match(/^\w+$/i)
    name = eval "#{record.type.to_s}"
    return true if record.variable.is_a? name
    record.errors.add :type, "not a kind of given class #{record.type}"
  else
    record.errors.add :type ,"invalid input detected"
  end
end
